    protected Paragraph( PAPX papx, Range parent, int start )
    {
        super( Math.max( parent._start, start ), Math.min( parent._end,
                papx.getEnd() ), parent );
        _props = papx.getParagraphProperties( _doc.getStyleSheet() );
        _papx = papx.getSprmBuf();
