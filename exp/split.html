     public CellRangeAddress[] getFormattingRanges(){
         ArrayList<CellRangeAddress> lst = new ArrayList<CellRangeAddress>();
         for (Object stRef : _cf.getSqref()) {
             String[] regions = stRef.toString().split(" ");
             for (int i = 0; i < regions.length; i++) {
                 lst.add(CellRangeAddress.valueOf(regions[i]));
             }
         }
         return lst.toArray(new CellRangeAddress[lst.size()]);
     }

    private static Artillery createArtillery(String string) {
        Iterable<String> split = split(string);
        int attack = Integer.parseInt(Iterables.get(split, 0));
        int health = Integer.parseInt(Iterables.get(split, 1));

        return new Artillery(attack, health);
    }

    public static List<String> splitStringIntoWords(String str, String[] excludedWords) {
        List<String> splittedStr = new ArrayList<String>(Arrays.asList(str.toLowerCase().trim().split("\\s+")));
        splittedStr.removeAll(Arrays.asList(excludedWords));
        for (int i=0; i<splittedStr.size(); i++)
        {
            splittedStr.set(i, splittedStr.get(i).replaceAll("[^а-яА-Яa-zA-Z0-9-]", ""));
        }
        return splittedStr;
    }
