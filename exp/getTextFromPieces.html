    public String getTextFromPieces()
    {
        String text = doc.getDocumentText();

        // Fix line endings (Note - won't get all of them
        text = text.replaceAll( "\r\r\r", "\r\n\r\n\r\n" );
        text = text.replaceAll( "\r\r", "\r\n\r\n" );

        if ( text.endsWith( "\r" ) )
        {
            text += "\n";
        }

        return text;
    }
