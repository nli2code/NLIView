    private void isCellLockedMatches(Locator loc1, Locator loc2) {
        boolean b1 = loc1.cell.getCellStyle().getLocked();
        boolean b2 = loc1.cell.getCellStyle().getLocked();
        if (b1 != b2) {
            addMessage(loc1, loc2,
                    "Cell Protection does not Match ::",
                (b1 ? "" : "NOT ")+"LOCKED",
                (b2 ? "" : "NOT ")+"LOCKED"
            );
        }
    }
