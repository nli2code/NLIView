    public byte[] getRgbWithTint() {
        byte[] rgb = ctColor.getRgb();
        if (rgb != null) {
            if(rgb.length == 4) {
               byte[] tmp = new byte[3];
               System.arraycopy(rgb, 1, tmp, 0, 3);
               rgb = tmp;
            }
            for (int i = 0; i < rgb.length; i++){
                rgb[i] = applyTint(rgb[i] & 0xFF, ctColor.getTint());
            }
        }
        return rgb;
    }
