    public void reorderSlide(XMLSlideShow ppt, int slideNumber, int newSlideNumber) {
        List<XSLFSlide> slides = ppt.getSlides();

        XSLFSlide secondSlide = slides.get(slideNumber);
        ppt.setSlideOrder(secondSlide, newSlideNumber);
