    public void testGetXlsPicture() {
        ExcelUtils excelUtils = new ExcelUtils();
        Workbook workbook = excelUtils.readExcel(ExcelUtils.class.getResourceAsStream("/excel/test.xls"), MIMETypeEnum.XLS);
        PictureData pictureData = excelUtils.getPictureData(workbook.getSheetAt(0), 5, 5);
        if (pictureData != null) {
            System.out.println(pictureData.getMimeType());
            ImageUtils.saveImage(pictureData.getData(), "b.png", MIMETypeEnum.PNG);
        } else {
            System.out.println("未读取到指定图片！");
        }
    }

Example 2

    public void testPictureDetectionWithPNG() throws Exception {
        HWPFDocument document = HWPFTestDataSamples.openSampleFile("PngPicture.doc");
        PicturesTable pictureTable = document.getPicturesTable();
        
        assertEquals(1, pictureTable.getAllPictures().size());
        
        Picture p = pictureTable.getAllPictures().get(0);
        assertEquals(PictureType.PNG, p.suggestPictureType());
        assertEquals("png", p.suggestFileExtension());
    }

Example 3

    public void setSignature(int signature) {
        switch (signature) {
            case 0x5420:
                uidInstanceCount = 1;
                break;
            case 0x5430:
                uidInstanceCount = 2;
                break;
            default:
                throw new IllegalArgumentException(signature+" is not a valid instance/signature value for PICT");
        }        
    }
