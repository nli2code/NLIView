Example 1

	public TextBox addTitle() {
		Placeholder pl = new Placeholder();
		pl.setShapeType(ShapeTypes.Rectangle);
		pl.getTextRun().setRunType(TextHeaderAtom.TITLE_TYPE);
		pl.setText("Click to edit title");
		pl.setAnchor(new java.awt.Rectangle(54, 48, 612, 90));
		addShape(pl);
		return pl;
	}

Example 2

    public HSLFTable createTable(int numRows, int numCols) {
        if (numRows < 1 || numCols < 1) {
            throw new IllegalArgumentException("numRows and numCols must be greater than 0");
        }
        HSLFTable s = new HSLFTable(numRows,numCols);
        // anchor is set in constructor based on numRows/numCols
        addShape(s);
        return s;
    }

Example 3

    public void fill(Shape shape){
        Path2D.Double path = new Path2D.Double(_transform.createTransformedShape(shape));
        HSLFFreeformShape p = new HSLFFreeformShape(_group);
        p.setPath(path);
        applyPaint(p);
        p.setLineColor(null);   //Fills must be "No Line"
        _group.addShape(p);
    }
