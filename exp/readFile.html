        SlideShow ppt = new SlideShow(slTests.openResourceAsStream("sound.ppt"));

        SoundData[] sound = ppt.getSoundData();
        assertEquals("Expected 1 sound", 1, sound.length);

        assertTrue(Arrays.equals(ref_data, sound[0].getData()));
