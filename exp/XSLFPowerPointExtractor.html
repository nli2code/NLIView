	public static void main(String[] args) throws Exception {
		if(args.length < 1) {
			System.err.println("Use:");
			System.err.println("  PLCDumper <filename>");
			System.exit(1);
		}
		PLCDumper dump = new PLCDumper(
				new FileInputStream(args[0])
		);

		System.out.println("Dumping " + args[0]);
		dump.dumpPLC();
	}
