	private String dumpBytes(byte[] data, int offset, int len) {
		StringBuffer ret = new StringBuffer();
		for(int i=0; i<len; i++) {
			int j = i + offset;
			int b = data[j];
			if(b < 0) { b += 256; }

			String bs = Integer.toHexString(b);
			if(bs.length() == 1)
				ret.append('0');
			ret.append(bs);
			ret.append(' ');
		}
		return ret.toString();
	}
