Example 1

	private Document getDocRecord() {
		Record[] r = ss.getRecords();
		for (int i = (r.length - 1); i >= 0; i--) {
			if (r[i] instanceof Document) {
				return (Document) r[i];
			}
		}
		throw new IllegalStateException("No Document record found");
	}
