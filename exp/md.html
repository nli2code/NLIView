	public static BigInteger getNearby(NormalisedDecimal md, int offset) {
		BigInteger frac = md.composeFrac();
		int be = frac.bitLength() - 24 - 1;
		int sc = frac.bitLength() - 64;
		return getNearby(frac.shiftRight(sc), be, offset);
	}
