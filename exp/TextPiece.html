        public int compare( TextPiece textPiece, TextPiece textPiece1 )
        {
            if ( textPiece.getPieceDescriptor().fc > textPiece1
                    .getPieceDescriptor().fc )
            {
                return 1;
            }
            else if ( textPiece.getPieceDescriptor().fc < textPiece1
                    .getPieceDescriptor().fc )
            {
                return -1;
            }
            else
            {
                return 0;
            }
        }

    public void dumpTextPieces( boolean withText )
    {
        for ( TextPiece textPiece : _doc.getTextTable().getTextPieces() )
        {
            System.out.println( textPiece );

            if ( withText )
            {
                System.out.println( "\t" + textPiece.getStringBuilder() );
            }
        }
    }
